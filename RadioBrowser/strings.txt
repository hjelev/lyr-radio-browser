# String tokens for the Radio Browser plugin.
# Lyrion looks up these tokens (via Slim::Utils::Strings) to render localized
# UI labels. Only EN is provided here; translators can add more languages by
# appending indented "<LANG>\t<text>" lines under each token.

PLUGIN_RADIOBROWSER
	EN	Radio Browser

PLUGIN_RADIOBROWSER_DESC
	EN	Browse and play internet radio stations from the community-driven Radio Browser directory (api.radio-browser.info).

PLUGIN_RADIOBROWSER_SEARCH
	EN	Search Stations

PLUGIN_RADIOBROWSER_SEARCH_PROMPT
	EN	Search for a station by name

PLUGIN_RADIOBROWSER_LOCAL
	EN	Local Stations

PLUGIN_RADIOBROWSER_RECENT
	EN	Recently Played

PLUGIN_RADIOBROWSER_CLEAR_RECENT
	EN	Clear recently played

PLUGIN_RADIOBROWSER_TOP
	EN	Top Stations

PLUGIN_RADIOBROWSER_TOP_VOTED
	EN	Most Voted

PLUGIN_RADIOBROWSER_TOP_CLICKED
	EN	Most Played

PLUGIN_RADIOBROWSER_BY_TAG
	EN	Tags

PLUGIN_RADIOBROWSER_SEARCH_TAGS
	EN	Search Tags

PLUGIN_RADIOBROWSER_BY_COUNTRY
	EN	By Country

PLUGIN_RADIOBROWSER_ERROR
	EN	Unable to load stations. Please try again later.

PLUGIN_RADIOBROWSER_NONE
	EN	No stations found

PLUGIN_RADIOBROWSER_PARTIAL
	EN	Showing partial results - the full list could not be loaded in time

PLUGIN_RADIOBROWSER_SETTINGS_COUNTRY
	EN	Country override

PLUGIN_RADIOBROWSER_SETTINGS_COUNTRY_DESC
	EN	Two-letter country code (ISO 3166-1 alpha-2, e.g. US, DE, BG) used for the Local Stations menu. Leave empty to auto-detect from your server's location.

PLUGIN_RADIOBROWSER_SETTINGS_MAXRESULTS
	EN	Maximum stations per list

PLUGIN_RADIOBROWSER_SETTINGS_MAXRESULTS_DESC
	EN	Maximum number of stations fetched for browse lists (Search, By Tag, By Country). Higher values show more stations but use more memory. Default: 5000.

PLUGIN_RADIOBROWSER_SETTINGS_TOPRESULTS
	EN	Maximum stations in Top charts

PLUGIN_RADIOBROWSER_SETTINGS_TOPRESULTS_DESC
	EN	Maximum number of stations fetched for the global Most Voted and Most Played charts. These are "top N" lists, so a small value keeps them fast; large values can make the chart slow to load or fail. Default: 100.

PLUGIN_RADIOBROWSER_SETTINGS_MAXTAGS
	EN	Maximum tags in list

PLUGIN_RADIOBROWSER_SETTINGS_MAXTAGS_DESC
	EN	Maximum number of tags shown in the Tags browse list. The Radio Browser directory has ~11 000 tags; higher values show more but may slow down the initial load. Default: 1000.

PLUGIN_RADIOBROWSER_SETTINGS_RECENTCOUNT
	EN	Recently played count

PLUGIN_RADIOBROWSER_SETTINGS_RECENTCOUNT_DESC
	EN	How many recently played stations to remember, most recent first. Re-playing a station moves it to the top. Range 1-1000. Default: 100.

PLUGIN_RADIOBROWSER_SETTINGS_CACHETTL
	EN	Cache lifetime (minutes)

PLUGIN_RADIOBROWSER_SETTINGS_CACHETTL_DESC
	EN	How long station lists are cached before being refetched from the Radio Browser API, in minutes. Higher values reduce API calls but make vote/play counts update less often. Default: 60.

PLUGIN_RADIOBROWSER_SETTINGS_HIDEBROKEN
	EN	Hide broken stations

PLUGIN_RADIOBROWSER_SETTINGS_HIDEBROKEN_DESC
	EN	Hide stations that failed Radio Browser's most recent reachability check. Uncheck to show all stations, including unverified or currently-broken ones. Enabled by default.

PLUGIN_RADIOBROWSER_SETTINGS_MENULOCATION
	EN	Menu location

PLUGIN_RADIOBROWSER_SETTINGS_MENULOCATION_DESC
	EN	Choose whether Radio Browser appears under the Radio menu or the Apps menu. Requires a Lyrion Music Server restart to take effect.

PLUGIN_RADIOBROWSER_SETTINGS_MENULOCATION_RADIO
	EN	Radio

PLUGIN_RADIOBROWSER_SETTINGS_MENULOCATION_APPS
	EN	Apps
